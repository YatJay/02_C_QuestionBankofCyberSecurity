求任意的一个m×m矩阵的最大数及其所在的行列数，m(2<=m<=20)及矩阵元素从键盘输入(只考虑int型)。

程序的运行效果应类似地如图所示，图中的5和
1 45 67 235 464
35 6 89 32342 8 
347 9346 47 95 746
46 785 436 19434 634
3235 567 24 8465 25
是从键盘输入的内容。

#include <stdio.h>

int main(void)
{
	int i, j, m, max, maxi, maxj;
	int array[20][20];

	printf("Please input m:");
	scanf("%d", &m);

	printf("\nPlease input array:\n");
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < m; j++)
		{
			scanf("%d", &array[i][j]);
		}
	}

	max = array[0][0];
maxi=0;maxj=0;
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < m; j++)
		{
			if (max < array[i][j])
			{
				max = array[i][j];
				maxi = i;
				maxj = j;
			}
		}
	}
	printf("Output:");
	printf("\nmax=%d,i=%d,j=%d\n", max, maxi, maxj);
	return 0;
}
